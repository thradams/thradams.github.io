const wchar_t*  IntToLiteral(int i)
{
    switch (i)
    {
    case '\0': return L"\\0";
    case '\r': return L"\\r";
    case '\n': return L"\\n";
    case '\t': return L"\\t";
    case '\'': return L"\\'";
    case '\"': return L"\\\"";
    case '\?': return L"\\?";
    case '\\': return L"\\\\";
    case '\a': return L"\\a";
    case '\b': return L"\\b";
    case '\f': return L"\\f";
    case '\v': return L"\\v";
    }

    static wchar_t ch[2] = {0,0};
    ch[0] = wchar_t(i);
    return ch;
}
