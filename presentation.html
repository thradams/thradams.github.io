Local and Literal functions for C2Y

Motivation

void async(void (*callback)(int result, void* data), void* data);

struct main_capture {
    int value;
};

static void main_async_complete(int result, void* data) {
    struct main_capture* capture = data;
    free(capture);
}

int main() {
    struct main_capture* capture = calloc(1, sizeof * capture);
    async(main_async_complete, capture);
}
  • Hard to read (ugly, unnecessary names)
  • Hard to check if void* cast is safe
  • Hard to maintain (is maincapture or mainasync_complete used by someone else? )

Motivation - do this, then that ...


void do_this_async(void (*callback)(int result, void* data), void* data);
void then_that_async(void (*callback)(int result, void* data), void* data);

struct do_this_capture {
    int value;
};

struct then_that_capture {
    char ch;
};

static void then_that_complete(int result, void* data);

static void do_this_complete(int result, void* data) {
    struct do_this_capture* capture1 = data;

    struct then_that_capture* capture2 = calloc(1, sizeof *capture2);
    then_that_async(then_that_complete, capture2);

    free(capture2);
}

static void then_that_complete(int result, void* data) {
    struct then_that_capture* capture2 = data;
    free(capture2);
}

int main() {
    struct do_this_capture* capture1 = calloc(1, sizeof *capture1);
    do_this_async(do_this_complete, capture1);
}
