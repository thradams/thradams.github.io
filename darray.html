static inline ptrarray_push(struct ptrarray* a, void* value)
{
    if (a->size + 1 > a->capacity)
    {
        int n = a->capacity * 2;
        if (n == 0)
        {
            n = 1;
        }
        void** pnew = a->data;
        pnew = (void**)realloc(pnew, n * sizeof(void*));
        if (pnew)
        {
            a->data = pnew;
            a->capacity = n;
        }
    }
    a->data[a->size] = value;
    a->size++;
}
