template<class It>
std::string to_string(It first, It last)
{
    std::string result;
    result += "{";

    size_t digits = last -  first ;
    for (int i = digits - 1; i >= 0; i--)
    {
        if (i != digits - 1)
        {
            result += " ";
        }

        result += std::to_string((unsigned long long)first[i]);
    }

    result += "}";
    return result;
}
